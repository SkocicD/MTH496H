
\newcommand{\nextindex}[3]{%
  % #1 = current index
  % #2 = maximum index
  % #3 = name of variable to store result
  \pgfmathtruncatemacro{\temp}{#1+1}%
  \ifnum\temp>#2
    \pgfmathtruncatemacro{#3}{1}%
  \else
    \pgfmathtruncatemacro{#3}{\temp}%
  \fi
}
